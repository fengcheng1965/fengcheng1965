```javascript
    console.log(link.href, link.textContent.trim());
});

// 检查死链接
links.forEach(link => {
    if (link.href === '#' || link.href === window.location.href) {
        console.warn('可能的死链接:', link);
    }
});
```

**自动化测试（可选）**
```javascript
// 使用JavaScript自动化测试
function testLinks() {
    const links = document.querySelectorAll('a');
    let passed = 0;
    let failed = 0;
    
    links.forEach(link => {
        if (link.href && link.href !== '#' && !link.href.includes('javascript:')) {
            fetch(link.href, { method: 'HEAD' })
                .then(response => {
                    if (response.ok) {
                        console.log('✅', link.href);
                        passed++;
                    } else {
                        console.error('❌', link.href, response.status);
                        failed++;
                    }
                })
                .catch(error => {
                    console.error('❌', link.href, error.message);
                    failed++;
                });
```
